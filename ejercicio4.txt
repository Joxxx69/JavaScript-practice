Algoritmo Matriz
    Definir n,filas, columnas como Entero
    Definir matriz como Entero
    Dimension matriz(100,100)

    Para filas=1 Hasta 5 Con Paso 1 Hacer 
        Para columnas=1 Hasta 5 Con Paso 1 Hacer
            Escribir "Ingrese valor para [",filas,"][",columnas,"]"
            leer matriz(filas,columnas)
        FinPara
    Fin Para 
    Para filas=1 Hasta n Con Paso 1 Hacer 
        Para columnas=1 Hasta n Con Paso 1 Hacer
            Si filas ==1 y columnas == 1 entonces
                Escribir "Esquina superior Izquierda [",filas,"][",columnas,"] =",matriz(filas,columnas) Sin saltar
            FinSi
            Si filas == 1 y columnas==5 entonces
                Escribir "Esquina superior Derecha [",filas,"][",columnas,"] =",matriz(filas,columnas) Sin saltar
            FinSi
            Si filas == 5 y columnas==1 entonces
                Escribir "Esquina inferior Izquierda [",filas,"][",columnas,"] =",matriz(filas,columnas) Sin saltar
            FinSi
            Si filas == 5 y columnas ==5 entonces
                Escribir "Esquina inferior Derecha [",filas,"][",columnas,"] =",matriz(filas,columnas) Sin saltar
            SiNo 
                Escribir "[",filas,"][",columnas,"] =",matriz(filas,columnas) Sin saltar
        FinPara
        Escribir " "

    Fin Para 


FinAlgoritmo
